#include <thread>   // C++ v11
#include <iostream>
#include <string>

using namespace std;

void show()
{
    for (int i = 1; i <= 1000; i++)
    {
        cout << "A" + to_string(i) + " ";
    }
}

void jump()
{
    for (int i = 1; i <= 500; i++)
    {
        cout << "B" + to_string(i) + " ";
    }
}

int main()  // Main thread is always a foreground
{
    // show();
    // jump();
    thread t1(show);    // background thread
    thread t2(jump);    // background thread

    for (int i = 1; i <= 500; i++)
    {
        cout << "C" + to_string(i) + " ";
    }
}
